Создание слайдера SWIPER

HTML:

  • В head перед стилями поключить < link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/swiper@10/swiper-bundle.min.css">
  • Перед закрывающим тегом body подключить скрипт src="https://cdn.jsdelivr.net/npm/swiper@10/swiper-bundle.min.js"
  • Создать структуру слайдера: swiper > swiper-wrapper > swiper-slide*n
  • Если нужны стрелки, точки-переключатели, скроллбар, добавить соответствующие блоки

JS: 

let swiper = new Swiper('.класс слайдера', {
	// Параметры слайдера
});
Параметры слайдера:
  • Стрелки-переключатели: 
    navigation: {
	nextEl: ".класс стрелки-вперёд",
	prevEl: ".класс стрелки-назад",
},
  • Точки-переключатели: 
    pagination: {
	el: ".swiper-pagination",
	clickable: true, //возможность нажимать на точки
	dynamicBullets: true, //динамические точки-переключатели
	type: "progressbar", //в виде полосы прокрутки
	type: "fraction", //в виде 1/5, 2/5 итд

	renderBullet: function (index, className) {
	return '< span class="' + className + '">' + (index + 1) + "< />";
	}, //пользовательские стили точек
},
  • Полоса прокрутки: 
    scrollbar: {
el: ".swiper-scrollbar",
hide: true,
},
  • Вертикальное направление слайдера: 
    direction: "vertical",
  • Отступы между слайдами: 
    spaceBetween: 30,
  • Количество показываемых слайдов: 
    slidesPerView: N,
  • Выравнивание слайдов по центру: 
    centeredSlides: true,
  • Возможность листать колесом мыши: 
    mousewheel: true,
  • Свободное пролистывание: 
    freeMode: true,
  • Курсор - рука-перетаскиватель: 
    grabCursor: true,
  • Бесконечное перелистывание: 
    loop: true,
  • Эффект выцветания при переключении (fade): 
    effect: "fade",
  • Эффект кручения куба: 
    effect: "cube",
cubeEffect: {
	shadow: true,
	slideShadows: true,
	shadowOffset: 20,
	shadowScale: 0.94,
},
  • Эффект жалюзи: 
    effect: "coverflow",
grabCursor: true,
centeredSlides: true,
slidesPerView: "auto",
coverflowEffect: {
	rotate: 50,
	stretch: 0,
	depth: 100,
	modifier: 1,
	slideShadows: true,
}
  • Эффект вращения карточки: 
    effect: "flip",
  • Эффект колоды карт: 
    effect: "cards",
  • Автоперелистывание: 
    autoplay: {
delay: 2500,
disableOnInteraction: false,
},
  • Скорость перелистывания: 
    speed: 600,
  • Возможность увеличивать изображение двойным кликом: 
    zoom: true,
  • Адаптация с помощью брейкпойнтов: 
    breakpoints: {
	640: {
		slidesPerView: 2,
		spaceBetween: 20,
	},
	768: {
		slidesPerView: 4,
		spaceBetween: 40,
	},
	1024: {
		slidesPerView: 5,
		spaceBetween: 50,
	},
},
